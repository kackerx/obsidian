func (l *LinkedList) reverse() {
	newHead := NewNode(-1, nil)

	cur := l.Head.next
	for cur != nil {
		next := cur.next
		cur.next = newHead.next
		newHead.next = cur
		cur = next
	}

	l.Head = newHead
}
